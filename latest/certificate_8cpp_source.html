00021         
00022         try
00023         {
00024             char cb[1000] = {0};
00025             sib.pbData = (BYTE*)cb; 
00026             sib.cbData = 1000;
00027             wchar_t*    szSubject= L"CN=Certificate";
00028             if (!CertStrToNameW(CRYPT_ASN_ENCODING, szSubject,0,0,sib.pbData,&sib.cbData,NULL))
00029                 throw;
00030             wchar_t* pszKeyContainerName = L"Container";
00031             if (!CryptAcquireContextW(&hProv,pszKeyContainerName,MS_DEF_PROV_W,PROV_RSA_FULL,CRYPT_NEWKEYSET | CRYPT_MACHINE_KEYSET))
00032             {
00033                 hr = GetLastError();
00034                 if (GetLastError() == NTE_EXISTS)
00035                 {
00036                     if (!CryptAcquireContextW(&hProv,pszKeyContainerName,MS_DEF_PROV_W,PROV_RSA_FULL,CRYPT_MACHINE_KEYSET))
00037                     {
00038                         throw;
00039                     }
00040                 }
00041                 else
00042                     throw;
00043             }
00044             
00045             if (!CryptGenKey(hProv, AT_KEYEXCHANGE, CRYPT_EXPORTABLE, &hKey))
00046                 throw;
00047             
00048             CRYPT_KEY_PROV_INFO kpi = {0};
00049             kpi.pwszContainerName = pszKeyContainerName;
00050             kpi.pwszProvName = MS_DEF_PROV_W;
00051             kpi.dwProvType = PROV_RSA_FULL;
00052             kpi.dwFlags = CERT_SET_KEY_CONTEXT_PROP_ID;
00053             kpi.dwKeySpec = AT_KEYEXCHANGE;
00054             
00055             SYSTEMTIME et;
00056             GetSystemTime(&et);
00057             et.wYear += 1;
00058             
00059             CERT_EXTENSIONS exts = {0};
00060             p = CertCreateSelfSignCertificate(hProv,&sib,0,&kpi,NULL,NULL,&et,&exts);
00061             AX = CryptFindCertificateKeyProvInfo(p,CRYPT_FIND_MACHINE_KEYSET_FLAG,NULL) ;
00062         }
00063         catch(...)
00064         {
00065         }
00066         
00067         // cleanup.
00068         if (hKey)
00069             CryptDestroyKey(hKey);
00070         hKey = 0;
00071         if (hProv)
00072             CryptReleaseContext(hProv,0);
00073         hProv = 0;
00074         // return certificate.
00075         return (p);
